// 牛顿环
let ctx=document.getElementById('c').getContext('2d');

function draw(){
let angle=document.getElementById('angle').value;
let press=document.getElementById('press').value;

ctx.clearRect(0,0,400,400);

let scale=1-angle/100;

for(let i=1;i<12;i++){
let r=i*18*(6/press);

ctx.beginPath();
ctx.ellipse(200,200,r,r*scale,0,0,Math.PI*2);
ctx.stroke();
}
}

setInterval(draw,30);
</script>
